---
title: "Bridge"
api: "POST https://api.mintlify.com/bridge"
description: "For bridging."
---

<ParamField query="quote (optional)" type="boolean">
  If true, it will return the bridge quote without the transaction object and
  won't accrue any fees.
</ParamField>

### Body

<ParamField body="qty" type="number">
  The quantity of tokens to bridge.
</ParamField>

<ParamField body="srcChainId" type="number">
  The source chain's unique identifier.
</ParamField>

<ParamField body="dstChainId" type="number">
  The destination chain's unique identifier.
</ParamField>

<ParamField body="srcChainTokenAddress" type="string">
  The token address on the source chain.
</ParamField>

<ParamField body="dstChainTokenAddress" type="string">
  The token address on the destination chain.
</ParamField>

<ParamField body="fromAddress" type="string">
  The address from which the tokens will be bridged.
</ParamField>

<ParamField body="toAddress" type="string">
  The address to which the tokens will be bridged.
</ParamField>

<ParamField body="slippage (optional)" type="number">
  The acceptable slippage percentage. This is an optional parameter.
</ParamField>

### Response

If the `quote` query parameter is set to `false` or not provided, the response will be:

<ResponseField name="alternativeRoutes" type="array">
  The address of the token.
</ResponseField>

<ResponseField name="bestRoute" type="object">
<Expandable>

<ResponseField name="bridgeInfo" type="object">
  <Expandable>
    <ResponseField name="bridgeExplorer" type="string">
      Bridge explorer base url.
    </ResponseField>
    <ResponseField name="logoUri" type="string?">
      Logo URI for the bridge. This field is optional and might not always be
      provided.
    </ResponseField>
    <ResponseField name="name" type="string">
      Name of the bridge.
    </ResponseField>
  </Expandable>
</ResponseField>

    <ResponseField name="dstAmountEstimate" type="int">
      Estimate of the amount of tokens to be received on the destination chain.
    </ResponseField>


     <ResponseField name="srcChainAggregatorInfo" type="object">
      <Expandable>
          <ResponseField name="logoURI" type="string?">
            Logo URI for the source chain aggregator. This field is optional and might not always be provided.
          </ResponseField>
           <ResponseField name="name" type="string">
            Name of the aggregator.
          </ResponseField>
      </Expandable>
    </ResponseField>


    <ResponseField name="dstChainAggregatorInfo" type="object">
      <Expandable>
          <ResponseField name="logoURI" type="string?">
            Logo URI for the destination chain aggregator. This field is optional and might not always be provided.
          </ResponseField>
           <ResponseField name="name" type="string">
            Name of the aggregator.
          </ResponseField>
      </Expandable>
    </ResponseField>

     <ResponseField name="srcChainSwapDexs" type="array">
      <Expandable>
          <ResponseField name="proportion" type="string">
              Proportion of the swap dex.
          </ResponseField>
           <ResponseField name="name" type="string">
              Name of the source chain swap dex.
          </ResponseField>
      </Expandable>
    </ResponseField>

      <ResponseField name="dstChainSwapDexs" type="array">
      <Expandable>
          <ResponseField name="proportion" type="string">
              Proportion of the swap dex.
          </ResponseField>
           <ResponseField name="name" type="string">
              Name of the destination chain swap dex.
          </ResponseField>
      </Expandable>
    </ResponseField>

</Expandable>
   </ResponseField>
